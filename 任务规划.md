#### P1.4 世界管理器

```
P1.4.1 实现 WorldManager 主类
  文件: src/world/world_manager.py
  
  class WorldManager:
      def __init__(self, config_path: str):
          self.config = load_yaml(config_path)
          self.mines: List[MineZone] = []
          self.stations: Dict[str, WorkStation] = {}
          self.warehouse: Warehouse = None
          self.energy_source: EnergySource = None
          self.charging_dock: ChargingDock = None
          self.path_network: PathNetwork = None
          self.item_registry: ItemRegistry = ItemRegistry()
          self.sim_time: float = 0.0
          self.energy_balance: float = 0.0  # 总能量收支
          
      def build_world(self, sim_context):
          """在Isaac Sim中构建完整世界"""
          # 调用 P1.2 中所有组件的创建函数
          
      def step(self, dt: float):
          """每个仿真步更新世界状态"""
          self.sim_time += dt
          self.energy_balance += self.energy_source.get_current_output(self.sim_time) * dt
          # 更新所有工站状态
          for station in self.stations.values():
              station.step(dt)
              
      def get_world_state(self) -> Dict:
          """返回完整世界状态 (供Brain使用)"""
          return {
              "sim_time": self.sim_time,
              "energy_balance": self.energy_balance,
              "mine_remaining": {m.name: m.remaining for m in self.mines},
              "warehouse_inventory": self.warehouse.get_inventory(),
              "station_status": {k: v.get_status() for k,v in self.stations.items()},
          }
          
  验证: build_world() 后场景中所有元素可见且物理正常
```

  class Battery:
      def __init__(self, capacity_wh=500):
          self.capacity = capacity_wh
          self.current_level = capacity_wh  # 满电
          
      def consume(self, power_watts, dt_seconds):
          energy = power_watts * dt_seconds / 3600  # Wh
          self.current_level = max(0, self.current_level - energy)
          
      def charge(self, power_watts, dt_seconds):
          energy = power_watts * dt_seconds / 3600
          self.current_level = min(self.capacity, self.current_level + energy)
          
      @property
      def soc(self) -> float:  # State of Charge, 0~1
          return self.current_level / self.capacity
          
      @property  
      def is_critical(self) -> bool:
          return self.soc < 0.15

#### P5.1 工站基类

```
P5.1.1 实现 WorkStation 基类
  文件: src/workstation/base_station.py
  
  class WorkStation(ABC):
      def __init__(self, name, station_type, position, recipes):
          self.name = name
          self.station_type = station_type
          self.position = position
          self.recipes = {r.name: r for r in recipes if r.station_type == station_type}
          self.state = "idle"  # idle / waiting_input / processing / done / error
          self.input_buffer: Dict[str, int] = {}  # 已投入的物料
          self.output_buffer: List[Item] = []       # 产出物品
          self.current_recipe: Optional[Recipe] = None
          self.process_timer: float = 0.0
          self.input_port_pose: SE3 = ...   # 入料口位姿
          self.output_port_pose: SE3 = ...  # 出料口位姿
          
      def receive_input(self, item_type: str, quantity: int = 1) -> bool:
          """接收投入的物料"""
          self.input_buffer[item_type] = self.input_buffer.get(item_type,0) + quantity
          self._check_recipe_ready()
          return True
          
      def _check_recipe_ready(self):
          """检查是否有配方可以开始"""
          for recipe in self.recipes.values():
              if all(self.input_buffer.get(k,0) >= v 
                     for k,v in recipe.inputs.items()):
                  self.current_recipe = recipe
                  self.state = "processing"
                  self.process_timer = recipe.process_time
                  # 消耗输入
                  for k,v in recipe.inputs.items():
                      self.input_buffer[k] -= v
                  break
                  
      def step(self, dt: float):
          if self.state == "processing":
              self.process_timer -= dt
              if self.process_timer <= 0:
                  # 生产完成
                  for item_type, qty in self.current_recipe.outputs.items():
                      for _ in range(qty):
                          self.output_buffer.append(
                              ItemRegistry.create(item_type))
                  self.state = "done"
                  self.current_recipe = None
                  
      def collect_output(self) -> Optional[Item]:
          """取出一个产品"""
          if self.output_buffer:
              self.state = "idle" if len(self.output_buffer) == 1 else "done"
              return self.output_buffer.pop(0)
          return None
          
      def get_status(self) -> Dict:
          return {
              "name": self.name, "state": self.state,
              "input_buffer": dict(self.input_buffer),
              "output_count": len(self.output_buffer),
              "process_remaining": max(0, self.process_timer),
              "available_recipes": list(self.recipes.keys()),
          }
```

  文件: src/brain/strategic_planner.py
  
  class StrategicPlanner:
      def __init__(self, llm_client, recipe_graph, world_state_fn):
          self.llm = llm_client  # OpenAI API / 本地模型
          self.recipe_graph = recipe_graph
          self.get_world_state = world_state_fn
          
      def generate_master_plan(self, goal: str = "assemble_robot") -> TaskPlan:
          """
          生成完成目标所需的完整任务计划
          """
          world_state = self.get_world_state()
          
          prompt = f"""
          你是一个工厂规划AI。你需要规划机器人的完整行动序列。
          
          ## 目标
          制造一个 {goal}
          
          ## 制造配方 (依赖关系)
          {self.recipe_graph.to_text()}
          
          ## 当前世界状态
          - 矿区剩余: {world_state['mine_remaining']}
          - 仓库库存: {world_state['warehouse_inventory']}
          - 工站状态: {world_state['station_status']}
          - 机器人电量: {world_state['battery_soc']}
          - 已用时间: {world_state['sim_time']}
          
          ## 约束
          1. 机器人一次只能携带1个物品 (单手) 或2个物品 (双手)
          2. 电量低于15%必须去充电
          3. 工站同时只能执行一个配方
          4. 应尽量减少总时间 (并行利用等待时间)
          
          ## 输出格式
          返回JSON格式的任务列表:
          [
            {{"task_id": 1, "type": "mine", "target": "iron_ore", "quantity": 3}},
            {{"task_id": 2, "type": "deliver_to_station", "items": ["iron_ore","iron_ore","iron_ore"], "station": "smelter"}},
            {{"task_id": 3, "type": "start_processing", "station": "smelter", "recipe": "smelt_iron"}},
            {{"task_id": 4, "type": "wait_or_parallel", "description": "趁冶炼时去采更多矿"}},
            ...
          ]
          
          请给出最优计划。
          """
          
          response = self.llm.chat(prompt)
          plan = TaskPlan.parse(response)
          return plan
          
      def replan(self, current_plan, failure_info) -> TaskPlan:
          """当执行失败时重新规划"""
          prompt = f"""
          原计划: {current_plan.to_text()}
          执行失败: {failure_info}
          当前状态: {self.get_world_state()}
          请生成修正后的计划。
          """
          response = self.llm.chat(prompt)
          return TaskPlan.parse(response)
